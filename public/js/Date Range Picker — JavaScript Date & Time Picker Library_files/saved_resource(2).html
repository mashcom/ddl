                                                                                                                                                                        Ken Mayer karthickrajan • a month ago

                                                                                                                                                                        If you are looking for a single date picker, the code below works for me for the JavaScript, replace "testdate" with the id for your own input control (in two places):

                                                                                                                                                                        $(function()
                                                                                                                                                                        {
                                                                                                                                                                        /*
                                                                                                                                                                        Note: you have to set the name in two places, and a full copy of this code for
                                                                                                                                                                        each date you have on a form.
                                                                                                                                                                        */
                                                                                                                                                                        $('input[name="testdate"]').daterangepicker({
                                                                                                                                                                        autoUpdateInput: false,
                                                                                                                                                                        locale:
                                                                                                                                                                        {
                                                                                                                                                                        cancelLabel: 'Clear'
                                                                                                                                                                        },
                                                                                                                                                                        singleDatePicker: true,
                                                                                                                                                                        showDropdowns: true
                                                                                                                                                                        });

                                                                                                                                                                        $('input[name="testdate"]').on('apply.daterangepicker', function(ev, picker)
                                                                                                                                                                        {
                                                                                                                                                                        $(this).val(picker.startDate.format('MMMM D, YYYY'));
                                                                                                                                                                        });
